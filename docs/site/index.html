Developer
API Reference

Query trust scores programmatically. Sub-100ms responses, globally distributed.

# Query an agent's trust score
curl https://api.isnad.dev/v1/trust/openai/gpt-4o \
  -H "Authorization: Bearer $ISNAD_KEY" \
  -H "Accept: application/json"

# Response
{
  "agent": "openai/gpt-4o",
  "score": 0.94,
  "dimensions": {
    "identity": 0.97,
    "behavioral": 0.91,
    "attestation": 0.95,
    "network": 0.89
  },
  "verified": true,
  "chain_length": 14
}
import isnad

client = isnad.Client("your-api-key")

# Get trust score
result = client.verify("openai/gpt-4o")

print(f"Score: {result.score}")        # 0.94
print(f"Verified: {result.verified}")  # True

# Stream trust events
for event in client.stream("openai/gpt-4o"):
    print(event.type, event.score)
import { Isnad } from '@isnad/sdk';

const client = new Isnad('your-api-key');

// Get trust score
const result = await client.verify('openai/gpt-4o');

console.log(result.score);     // 0.94
console.log(result.verified);  // true
console.log(result.dimensions);
// { identity: 0.97, behavioral: 0.91,
//   attestation: 0.95, network: 0.89 }
